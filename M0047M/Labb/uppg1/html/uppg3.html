k_values = 1:100;
a = zeros(size(k_values));

for i = 1:length(k_values)
    k = k_values(i);
    a(i) = exp(k * log((k - 3) / k));
end

figure;
plot(k_values, a, 'bo', 'MarkerSize', 4);
hold on;
xlabel('k');
ylabel('a_k');
grid on;
axis([0 100, 0 0.05]);
hold off;
